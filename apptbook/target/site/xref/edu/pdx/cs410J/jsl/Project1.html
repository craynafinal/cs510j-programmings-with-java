7   /**
8    * The main class for the CS410J appointment book Project.
9    * This class will receive command line arguments to create each instance of
10   * the {@link Appointment} and the {@link AppointmentBook} classes, and it is going to add
11   * an appointment to an appointment book. To do this, it will parse the
12   * command line arguments and will interpret argument data and options.
13   *
14   * @author Jong Seong Lee
15   * @version   %I%, %G%
16   * @since     1.0
17   */
18  public class Project1 {
19    private static final int MAX_ARGUMENTS = 4;
20    private static final String[] allowed_options = { "-print", "-README" };
21  
22    /**
23     * This method will print a given error message
24     * and will terminate the program execution with the exit code of 1.
25     *
26     * @param message a message of an error in string format
27       */
28    private static void programFail(String message) {
29      System.err.println(message);
30      System.exit(1);
31    }
32  
33    /**
34     * This method will check if a given command line argument is a recognizable option.
35     *
36     * @param option  a command line argument in string format
37     * @return        true if a command line option is recognized, otherwise false
38       */
39    private static boolean isOption(String option) {
40      return Arrays.asList(allowed_options).contains(option);
41    }
42  
43    /**
44     * This method will use regex to check if a given date is in date format.
45     *
46     * @param date  a date in string format
47     * @return      true if a date is in date format, otherwise false
48       */
49    private static boolean isDateFormatCorrect(String date) {
50      return date.matches("(0?[1-9]|1[012])/(0?[1-9]|[12][0-9]|3[01])/[0-9]{4,4} ([01]?[0-9]|2[0-3]):[0-5]?[0-9]");
51    }

87    /**
88     * This main method will parse the command line arguments and will
89     * create each instance of the {@link Appointment} and the {@link AppointmentBook} classes
90     * to add an appointment to an appointment book.
91     *
92     * Arguments for an appointment and an appointment book must be provided in order.
93     * They are a name of owner and a description, a begin time, and an end time of an appointment.
94     * The first two can be any string, but the last two has to be in date format.
95     *
96     * When -print option is given, it will print information regarding
97     * an appointment which is created by arguments passed, description and begin / end times.
98     *
99     * When -README option is given, it will print readme text of this project.
100    * This will explain what this program does and usage of this program.
101    *
102    * Following is the usage description from the project instruction document:
103    *
104    * Usage: java -jar target/apptbook-1.0-SNAPSHOT.jar [options] <args>
105    *   args are (in this order):
106    *   owner                    The person whose owns the appt book
107    *   description              A description of the appointment
108    *   beginTime                When the appt begins (24-hour time)
109    *   endTime                  When the appt ends (24-hour time)
110    * options are (options may appear in any order):
111    *   -print                   Prints a description of the new appointment
112    *   -README                  Prints a README for this project and exits
113    * Date and time should be in the format: mm/dd/yyyy hh:mm
114    *
115    * @param args  a name of owner, a description, a begin time, a end time,
116    *              and options -print, and -README
117    *              those time arguments should be in date format of mm/dd/yyyy hh:mm
118    *              year must be 4 digits and other fields may be 1 or 2 digits
119      */
120   public static void main(String[] args) {
121     List<String> arguments = new ArrayList<String>();
122     List<String> options = new ArrayList<String>();
123     Appointment appointment = null;
124     AppointmentBook appointment_book = null;
125 
126     for (int i = 0; i < args.length; i++) {
127       switch (args[i].charAt(0)) {
128         case '-':
129           if (isOption(args[i])) {
130             options.add(args[i]);
131           } else {
132             programFail("Option " + args[i] + " is not recognized");
133           }
134           break;
135         default:
136           arguments.add(args[i]);
137           break;
138       }
139     }
140 
141     if (options.contains(allowed_options[1])) {
142       printReadMe();
143     } else {
144       // number of arguments check
145       if (arguments.size() < MAX_ARGUMENTS) {
146         programFail("Missing command line arguments");
147       } else if (arguments.size() > MAX_ARGUMENTS) {
148         programFail("Too many command line arguments");
149       }
150 
151       // date format check
152       for (String date : arguments.subList(2, 4)) {
153         if (!isDateFormatCorrect(date)) {
154           programFail("Argument " + date + " is not in date format");
155         }
156       }
157 
158       appointment_book = new AppointmentBook(arguments.get(0));
159       appointment = new Appointment(arguments.subList(1, arguments.size()));
160       appointment_book.addAppointment(appointment);
161 
162       if (options.contains(allowed_options[0])) {
163         System.out.println(appointment);
164       }
165     }
166   }
