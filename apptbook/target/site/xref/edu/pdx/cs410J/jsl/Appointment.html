1   package edu.pdx.cs410J.jsl;
2   
3   import edu.pdx.cs410J.AbstractAppointment;
4   import java.util.List;
5   
6   /**
7    * The <code>Appointment</code> is the class extended from the {@link AbstractAppointment} class
8    * that describes an appointment with description, begin time and end time.
9    *
10   * @author    Jong Seong Lee
11   * @version   %I%, %G%
12   * @since     1.0
13   */
14  public class Appointment extends AbstractAppointment {
15    private String description = null;
16    private String begin_time = null;
17    private String end_time = null;
18  
19    /**
20     * This constructor takes no parameter and it uses default values to initialize.
21     */
22    public Appointment() {
23      this("description", "01/01/2000 00:00", "01/01/2000 00:00");
24    }
25  
26    /**
27     * This constructor takes a list of arguments as a parameter, and the list should have
28     * description, begin time, and end time in order as <code>String</code> data type.
29     *
30     * @param arguments   a List collection that contains description, begin time and end time in order
31       */
32    public Appointment(List<String> arguments) {
33      this(arguments.get(0), arguments.get(1), arguments.get(2));
34    }
35  
36    /**
37     * This constructor takes description, begin time and end time in order.
38     *
39     * @param desc    a description of an appointment in string format
40     * @param begin   a begin time of an appointment in string format
41     * @param end     an end time of an appointment in string format
42       */
43    public Appointment(String desc, String begin, String end) {
44      description = desc;
45      begin_time = begin;
46      end_time = end;
47    }
48  
49    /**
50     * Returns a begin time of an instance in string format.
51     *
52     * @return  a begin time in string format
53       */
54    @Override
55    public String getBeginTimeString() {
56      return begin_time;
57    }
58  
59    /**
60     * Returns an end time of an instance in string format.
61     *
62     * @return  an end time in string format
63       */
64    @Override
65    public String getEndTimeString() {
66      return end_time;
67    }
68  
69    /**
70     * Returns a description of an instance in string format.
71     *
72     * @return  a description in string format
73       */
74    @Override
75    public String getDescription() {
76      return description;
77    }
78  }
