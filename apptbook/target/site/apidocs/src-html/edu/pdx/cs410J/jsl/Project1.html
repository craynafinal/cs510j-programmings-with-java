001package edu.pdx.cs410J.jsl;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.List;
006
007/**
008 * The main class for the CS410J appointment book Project.
009 * This class will receive command line arguments to create each instance of
010 * the {@link Appointment} and the {@link AppointmentBook} classes, and it is going to add
011 * an appointment to an appointment book. To do this, it will parse the
012 * command line arguments and will interpret argument data and options.
013 *
014 * @author Jong Seong Lee
015 * @version   %I%, %G%
016 * @since     1.0
017 */
018public class Project1 {
019  private static final int MAX_ARGUMENTS = 4;
020  private static final String[] allowed_options = { "-print", "-README" };
021
022  /**
023   * This method will print a given error message
024   * and will terminate the program execution with the exit code of 1.
025   *
026   * @param message a message of an error in string format
027     */
028  private static void programFail(String message) {
029    System.err.println(message);
030    System.exit(1);
031  }
032
033  /**
034   * This method will check if a given command line argument is a recognizable option.
035   *
036   * @param option  a command line argument in string format
037   * @return        true if a command line option is recognized, otherwise false
038     */
039  private static boolean isOption(String option) {
040    return Arrays.asList(allowed_options).contains(option);
041  }
042
043  /**
044   * This method will use regex to check if a given date is in date format.
045   *
046   * @param date  a date in string format
047   * @return      true if a date is in date format, otherwise false
048     */
049  private static boolean isDateFormatCorrect(String date) {
050    return date.matches("(0?[1-9]|1[012])/(0?[1-9]|[12][0-9]|3[01])/[0-9]{4,4} ([01]?[0-9]|2[0-3]):[0-5]?[0-9]");
051  }
052
053  /**
054   * This method will print the readme information to standard output.
055   */
056  private static void printReadMe() {
057    System.out.println(
058            "********************************************************\n" +
059            "CS410/510J Advanced Java Programming\n" +
060            "Project 1: Designing an Appointment Book Application\n" +
061            "Student: Jong Seong Lee\n" +
062            "********************************************************\n" +
063            "\n" +
064            "Usage: java -jar target/apptbook-1.0-SNAPSHOT.jar [options] <args>\n" +
065            "  args are (in this order):\n" +
066            "    owner                    The person whose owns the appt book\n" +
067            "    description              A description of the appointment\n" +
068            "    beginTime                When the appt begins (24-hour time)\n" +
069            "    endTime                  When the appt ends (24-hour time)\n" +
070            "  options are (options may appear in any order):\n" +
071            "    -print                   Prints a description of the new appointment\n" +
072            "    -README                  Prints a README for this project and exits\n" +
073            "  Date and time should be in the format: mm/dd/yyyy hh:mm\n" +
074            "\n" +
075            "This program will create an appointment book that belongs to an owner;" +
076            "the owner’s name is given in the command line argument. " +
077            "Then it is going to create an appointment to be added in the appointment book that is created. " +
078            "For this, it will take three command line arguments: " +
079            "description, beginTime, and endTime to describe an appointment.\n" +
080            "\n" +
081            "There are two options available. To print an information regarding an appointment created, " +
082            "type -print, then the program will provide an appointment’s description, beginTime, " +
083            "and endTime on the terminal screen. If -README is provided, nothing will happen " +
084            "but printing this information including program usage and description on the terminal screen.");
085  }
086
087  /**
088   * This main method will parse the command line arguments and will
089   * create each instance of the {@link Appointment} and the {@link AppointmentBook} classes
090   * to add an appointment to an appointment book.
091   *
092   * Arguments for an appointment and an appointment book must be provided in order.
093   * They are a name of owner and a description, a begin time, and an end time of an appointment.
094   * The first two can be any string, but the last two has to be in date format.
095   *
096   * When -print option is given, it will print information regarding
097   * an appointment which is created by arguments passed, description and begin / end times.
098   *
099   * When -README option is given, it will print readme text of this project.
100   * This will explain what this program does and usage of this program.
101   *
102   * Following is the usage description from the project instruction document:
103   *
104   * Usage: java -jar target/apptbook-1.0-SNAPSHOT.jar [options] <args>
105   *   args are (in this order):
106   *   owner                    The person whose owns the appt book
107   *   description              A description of the appointment
108   *   beginTime                When the appt begins (24-hour time)
109   *   endTime                  When the appt ends (24-hour time)
110   * options are (options may appear in any order):
111   *   -print                   Prints a description of the new appointment
112   *   -README                  Prints a README for this project and exits
113   * Date and time should be in the format: mm/dd/yyyy hh:mm
114   *
115   * @param args  a name of owner, a description, a begin time, a end time,
116   *              and options -print, and -README
117   *              those time arguments should be in date format of mm/dd/yyyy hh:mm
118   *              year must be 4 digits and other fields may be 1 or 2 digits
119     */
120  public static void main(String[] args) {
121    List<String> arguments = new ArrayList<String>();
122    List<String> options = new ArrayList<String>();
123    Appointment appointment = null;
124    AppointmentBook appointment_book = null;
125
126    for (int i = 0; i < args.length; i++) {
127      switch (args[i].charAt(0)) {
128        case '-':
129          if (isOption(args[i])) {
130            options.add(args[i]);
131          } else {
132            programFail("Option " + args[i] + " is not recognized");
133          }
134          break;
135        default:
136          arguments.add(args[i]);
137          break;
138      }
139    }
140
141    if (options.contains(allowed_options[1])) {
142      printReadMe();
143    } else {
144      // number of arguments check
145      if (arguments.size() < MAX_ARGUMENTS) {
146        programFail("Missing command line arguments");
147      } else if (arguments.size() > MAX_ARGUMENTS) {
148        programFail("Too many command line arguments");
149      }
150
151      // date format check
152      for (String date : arguments.subList(2, 4)) {
153        if (!isDateFormatCorrect(date)) {
154          programFail("Argument " + date + " is not in date format");
155        }
156      }
157
158      appointment_book = new AppointmentBook(arguments.get(0));
159      appointment = new Appointment(arguments.subList(1, arguments.size()));
160      appointment_book.addAppointment(appointment);
161
162      if (options.contains(allowed_options[0])) {
163        System.out.println(appointment);
164      }
165    }
166  }
167
168}
