001package edu.pdx.cs410J.jsl;
002
003import edu.pdx.cs410J.AbstractAppointment;
004import java.util.List;
005
006/**
007 * The <code>Appointment</code> is the class extended from the {@link AbstractAppointment} class
008 * that describes an appointment with description, begin time and end time.
009 *
010 * @author    Jong Seong Lee
011 * @version   %I%, %G%
012 * @since     1.0
013 */
014public class Appointment extends AbstractAppointment {
015  private String description = null;
016  private String begin_time = null;
017  private String end_time = null;
018
019  /**
020   * This constructor takes no parameter and it uses default values to initialize.
021   */
022  public Appointment() {
023    this("description", "01/01/2000 00:00", "01/01/2000 00:00");
024  }
025
026  /**
027   * This constructor takes a list of arguments as a parameter, and the list should have
028   * description, begin time, and end time in order as <code>String</code> data type.
029   *
030   * @param arguments   a List collection that contains description, begin time and end time in order
031     */
032  public Appointment(List<String> arguments) {
033    this(arguments.get(0), arguments.get(1), arguments.get(2));
034  }
035
036  /**
037   * This constructor takes description, begin time and end time in order.
038   *
039   * @param desc    a description of an appointment in string format
040   * @param begin   a begin time of an appointment in string format
041   * @param end     an end time of an appointment in string format
042     */
043  public Appointment(String desc, String begin, String end) {
044    description = desc;
045    begin_time = begin;
046    end_time = end;
047  }
048
049  /**
050   * Returns a begin time of an instance in string format.
051   *
052   * @return  a begin time in string format
053     */
054  @Override
055  public String getBeginTimeString() {
056    return begin_time;
057  }
058
059  /**
060   * Returns an end time of an instance in string format.
061   *
062   * @return  an end time in string format
063     */
064  @Override
065  public String getEndTimeString() {
066    return end_time;
067  }
068
069  /**
070   * Returns a description of an instance in string format.
071   *
072   * @return  a description in string format
073     */
074  @Override
075  public String getDescription() {
076    return description;
077  }
078}
