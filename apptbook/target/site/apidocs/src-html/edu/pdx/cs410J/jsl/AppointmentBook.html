001package edu.pdx.cs410J.jsl;
002
003import edu.pdx.cs410J.AbstractAppointmentBook;
004import java.util.ArrayList;
005import java.util.List;
006
007/**
008 * This <code>AppointmentBook</code> class describes who owns a list of appointments
009 * and his / her list of appointments. For appointments it uses the {@link Appointment} class
010 * to represent appointments owned by an instance of appointmentBook.
011 *
012 * @author Jong Seong Lee
013 * @version   %I%, %G%
014 * @since     1.0
015 */
016public class AppointmentBook extends AbstractAppointmentBook<Appointment> {
017    private String owner_name = null;
018    private List<Appointment> appointments = null;
019
020    /**
021     * Initializes an instance with a default name of owner
022     */
023    public AppointmentBook() {
024        this("default name");
025    }
026
027    /**
028     * Initialize an instance with a given name of owner
029     * @param owner     a name of an owner of an appointment book in string format
030     */
031    public AppointmentBook(String owner) {
032        owner_name = owner;
033        appointments = new ArrayList<Appointment>();
034    }
035
036    /**
037     * Returns a name of an owner
038     * @return  a name of an owner in string format
039     */
040    @Override
041    public String getOwnerName() { return owner_name; }
042
043    /**
044     * Returns a list of appointments
045     * @return  a <code>List</code> collection of appointments
046     */
047    @Override
048    public List<Appointment> getAppointments() {
049        return appointments;
050    }
051
052    /**
053     * Adds a given appointment to a list of appointments
054     * @param appointment   an instance of the {@link Appointment} class
055     */
056    @Override
057    public void addAppointment(Appointment appointment) {
058        appointments.add(appointment);
059    }
060}
