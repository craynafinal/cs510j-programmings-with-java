001package edu.pdx.cs410J.jsl;
002
003import edu.pdx.cs410J.jsl.AppointmentBook;
004import org.hamcrest.Matcher;
005import org.junit.Test;
006
007import static org.hamcrest.CoreMatchers.*;
008import static org.hamcrest.MatcherAssert.assertThat;
009
010/**
011 * Unit tests for the {@link AppointmentBook} class.
012 */
013public class AppointmentBookTest {
014    /**
015     * The constructor of the AppointmentBookTest class.
016     */
017    public AppointmentBookTest() {
018    }
019
020    /**
021     * It will check if a name of an owner assigned correctly.
022     */
023    @Test
024    public void getOwnerNameShouldBeImplemented() {
025        String owner = "owner name";
026        AppointmentBook appointmentBook = new AppointmentBook(owner);
027        assertThat(appointmentBook.getOwnerName(), is(equalTo(owner)));
028    }
029
030    /**
031     * It will check if a list of appointments is empty initially.
032     */
033    @Test
034    public void getAppointmentShouldReturnEmptyWithoutAdd() {
035        AppointmentBook appointmentBook = new AppointmentBook();
036        assertThat(appointmentBook.getAppointments().size(), is(equalTo(0)));
037    }
038
039    /**
040     * It will check if an appointment is added correctly to a list of appointments.
041     */
042    @Test
043    public void getAppointmentShouldReturnAppointment() {
044        AppointmentBook appointmentBook = new AppointmentBook();
045        Appointment appointment = new Appointment();
046        appointmentBook.addAppointment(appointment);
047        assertThat(appointmentBook.getAppointments(), (Matcher)hasItems(appointment));
048    }
049}
