001package edu.pdx.cs410J.jsl;
002
003import org.junit.Test;
004
005import static org.hamcrest.CoreMatchers.*;
006import static org.hamcrest.MatcherAssert.assertThat;
007
008/**
009 * Unit tests for the {@link Appointment} class.
010 */
011public class AppointmentTest {
012
013  /**
014   * It will check if a description of an appointment is assigned correctly.
015   */
016  @Test
017  public void getDescriptionNeedsToBeImplemented() {
018    String description = "test description";
019
020    Appointment appointment = new Appointment(description, "", "");
021    assertThat(appointment.getDescription(), is(equalTo(description)));
022  }
023
024  /**
025   * It will check if a begin time of an appointment is assigned correctly.
026   */
027  @Test
028  public void getBeginTimeStringNeedsToBeImplemented() {
029    String begin_time = "11/11/1111 11:11";
030    Appointment appointment = new Appointment("", begin_time, "");
031    assertThat(appointment.getBeginTimeString(), is(equalTo(begin_time)));
032  }
033
034  /**
035   * It will check if an end time of an appointment is assigned correctly.
036   */
037  @Test
038  public void getEndTimeStringNeedsToBeImplemented() {
039    String end_time = "11/11/1111 11:11";
040    Appointment appointment = new Appointment("", "", end_time);
041    assertThat(appointment.getEndTimeString(), is(equalTo(end_time)));
042  }
043
044  /**
045   * It will check if toString method of the {@link Appointment} class works correctly.
046   */
047  @Test
048  public void toStringShouldContainCorrectDescriptionAndTimes() {
049    String description = "test description";
050    String begin_time = "11/11/1111 11:11";
051    String end_time = "22/22/2222 22:22";
052
053    Appointment appointment = new Appointment(description, begin_time, end_time);
054    assertThat(appointment.toString(), is(equalTo(description + " from " + begin_time + " until " + end_time)));
055  }
056
057  /**
058   * getBeginTime method is okay to return null for the project 1
059   */
060  @Test
061  public void forProject1ItIsOkayIfGetBeginTimeReturnsNull() {
062    Appointment appointment = new Appointment();
063    assertThat(appointment.getBeginTime(), is(nullValue()));
064  }
065
066}
